Getting Started
Up and running in 30 seconds.
No npm. No bundler. Just a script tag.
<!-- 1. include the script --> <script src="kaapi.js"></script> <!-- 2. use classes on any element --> <div class="kaapi-flex kaapi-gap-16 kaapi-p-24 kaapi-rounded-md"> <p class="kaapi-text-xl kaapi-font-bold kaapi-text-purple">Hello, Kaapi</p> </div> <!-- 3. for dynamically added elements --> <script> window.kaapi.apply(newElement); </script>
